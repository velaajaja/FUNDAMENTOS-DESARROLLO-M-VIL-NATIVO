//Imprimir todos los múltiplos de 3 que hay entre 1 y 100.
fun main() {
    println("Múltiplos de 3 entre 1 y 100:")
    for (i in 1..100) {
        if (i % 3 == 0) {
            println(i)
        }
    }
}

//Escribir un programa que sume los cuadrados de los cien primeros números naturales, mostrando el resultado en pantalla.
fun main() {
    var sumaCuadrados = 0
    for (i in 1..100) {
        val cuadrado = i * i
        sumaCuadrados += cuadrado
    }
    println("La suma de los cuadrados de los cien primeros números naturales es: $sumaCuadrados")
}
6.)
//Dados dos números naturales, el primero menor que el segundo, generar y mostrar todos los números comprendidos entre ellos en secuencia ascendente.
fun main() {
    println("Ingrese el primer número:")
    val num1 = readLine()!!.toInt()
    println("Ingrese el segundo número:")
    val num2 = readLine()!!.toInt()

    if (num1 < num2) {
        println("Números entre $num1 y $num2 en secuencia ascendente:")
        for (i in num1..num2) {
            println(i)
        }
    } else {
        println("El primer número debe ser menor que el segundo.")
    }
}
7.)
//Sumar todos los números que se digitan por teclado mientras no sea cero.
fun main() {
    var suma = 0
    var numero: Int

    do {
        println("Ingrese un número (ingrese 0 para terminar):")
        numero = readLine()!!.toInt()
        suma += numero
    } while (numero != 0)

    println("La suma de los números ingresados es: $suma")
}
